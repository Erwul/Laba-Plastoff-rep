5 основных паттернов JavaScript

Паттерны проектирования - это проверенные решения распространенных проблем в разработке программного обеспечения. Изучите основные паттерны JavaScript с примерами реализации.

Модуль (Module)

Инкапсулирует логику в самодостаточные модули, скрывая внутренние детали реализации и предоставляя публичный API. Помогает организовать код и избегать загрязнения глобального пространства имен.

// Модуль с помощью замыкания
const CounterModule = (function() {
  let count = 0;
  
  const increment = () => {
    count++;
    return count;
  };
  
  const decrement = () => {
    count--;
    return count;
  };
  
  const getCount = () => count;
  
  // Публичный API
  return {
    increment,
    decrement,
    getCount
  };
})();
Одиночка (Singleton)

Гарантирует, что класс имеет только один экземпляр и предоставляет глобальную точку доступа к нему. Используется для управления общими ресурсами, такими как подключения к базе данных или настройки приложения.

// Singleton с помощью замыкания
const ConfigManager = (function() {
  let instance = null;
  let config = {};
  
  function createInstance() {
    return {
      set: (key, value) => {
        config[key] = value;
      },
      get: (key) => config[key],
      getAll: () => ({...config})
    };
  }
  
  return {
    getInstance: function() {
      if (!instance) {
        instance = createInstance();
      }
      return instance;
    }
  };
})();
Фабрика (Factory)

Создает объекты без указания конкретного класса создаваемого объекта. Фабричный метод определяет интерфейс для создания объекта, но позволяет подклассам изменять тип создаваемых объектов.

// Фабрика для создания разных типов пользователей
class UserFactory {
  createUser(type, name) {
    switch(type) {
      case 'admin':
        return new Admin(name);
      case 'moderator':
        return new Moderator(name);
      case 'member':
        return new Member(name);
      default:
        throw new Error('Неизвестный тип пользователя');
    }
  }
}

class Admin {
  constructor(name) {
    this.name = name;
    this.role = 'Администратор';
    this.permissions = ['read', 'write', 'delete', 'manage'];
  }
}
Наблюдатель (Observer)

Определяет зависимость "один-ко-многим" между объектами так, что при изменении состояния одного объекта все зависящие от него объекты уведомляются и обновляются автоматически.

// Простая реализация паттерна Наблюдатель
class Observable {
  constructor() {
    this.observers = [];
  }
  
  subscribe(observer) {
    this.observers.push(observer);
  }
  
  unsubscribe(observer) {
    this.observers = this.observers.filter(obs => obs !== observer);
  }
  
  notify(data) {
    this.observers.forEach(observer => observer.update(data));
  }
}

class Observer {
  update(data) {
    console.log(`Получены данные: ${data}`);
  }
}
Стратегия (Strategy)

Определяет семейство алгоритмов, инкапсулирует каждый из них и делает их взаимозаменяемыми. Позволяет изменять алгоритмы независимо от клиентов, которые их используют.

// Стратегии для различных способов оплаты
class PaymentContext {
  constructor(strategy) {
    this.strategy = strategy;
  }
  
  setStrategy(strategy) {
    this.strategy = strategy;
  }
  
  executePayment(amount) {
    return this.strategy.pay(amount);
  }
}

class CreditCardPayment {
  pay(amount) {
    return `Оплачено ${amount}₽ с помощью кредитной карты`;
  }
}

class PayPalPayment {
  pay(amount) {
    return `Оплачено ${amount}₽ через PayPal`;
  }
}

Почему паттерны важны?

Паттерны проектирования в JavaScript помогают решать распространенные задачи разработки проверенными способами. Они обеспечивают:

  • Читаемость кода: Единый подход к решению проблем
  • Поддерживаемость: Легче вносить изменения и расширять функционал
  • Переиспользуемость: Проверенные решения для повторяющихся задач
  • Коммуникацию: Общий язык для разработчиков

Используя паттерны, вы пишете более структурированный, предсказуемый и масштабируемый код.
